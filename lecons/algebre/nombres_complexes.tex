\secnum{navyblue}{1}{ReprĂŠsentations et opĂŠrations ĂŠlĂŠmentaires}

\begin{tcolorbox}[thmbox]
  \textbf{Forme algĂŠbrique.}\enspace%
  \index{nombres complexes!forme algĂŠbrique}
  Tout $z \in \mathbb{C}$ s'ĂŠcrit de faĂ§on unique
  \[
    z = a + ib, \quad a = \mathrm{Re}(z) \in \mathbb{R},\quad b = \mathrm{Im}(z) \in \mathbb{R}.
  \]
  Le \textbf{conjuguĂŠ}\index{nombres complexes!conjuguĂŠ} et le
  \textbf{module}\index{nombres complexes!module} sont :
  \[
    \bar{z} = a - ib, \qquad |z| = \sqrt{a^2 + b^2} = \sqrt{z\bar{z}}.
  \]
\end{tcolorbox}

\rubriq{PropriĂŠtĂŠs du conjuguĂŠ et du module}

Les rĂ¨gles suivantes dĂŠcoulent directement des dĂŠfinitions. Pour tous
$z, w \in \mathbb{C}$ :
\[
  \overline{z + w} = \bar{z} + \bar{w}, \quad
  \overline{zw} = \bar{z}\,\bar{w}, \quad
  \overline{\bar{z}} = z, \quad
  z + \bar{z} = 2\,\mathrm{Re}(z), \quad
  z - \bar{z} = 2i\,\mathrm{Im}(z).
\]
Pour le module :
\[
  |zw| = |z|\,|w|, \quad
  \left|\frac{z}{w}\right| = \frac{|z|}{|w|} \;(w \neq 0), \quad
  |\bar{z}| = |z|, \quad
  |z + w| \leq |z| + |w|.
\]

\rubriq{Inverse et division}
Pour $z = a + ib \neq 0$ :
\[
  z^{-1} = \frac{\bar{z}}{|z|^2} = \frac{a - ib}{a^2 + b^2}.
\]
La division se ramĂ¨ne Ă  une multiplication par le conjuguĂŠ du dĂŠnominateur :
$\dfrac{z}{w} = \dfrac{z\bar{w}}{|w|^2}$.

\secnum{navyblue}{2}{Forme trigonomĂŠtrique et formule d'Euler}

\begin{tcolorbox}[thmbox]
  \textbf{Forme trigonomĂŠtrique.}\enspace%
  \index{nombres complexes!forme trigonomĂŠtrique}\index{argument}
  Tout $z \neq 0$ s'ĂŠcrit
  \[
    z = |z|(\cos\theta + i\sin\theta), \quad \theta \in \mathbb{R}.
  \]
  Le rĂŠel $\theta$ est un \textbf{argument}\index{argument} de $z$, notĂŠ
  $\theta \equiv \arg(z) \pmod{2\pi}$. L'argument principal est choisi dans
  $]-\pi, \pi]$.

  \medskip
  \textbf{Formule d'Euler.}\enspace%
  \index{formule d'Euler}
  Pour tout $\theta \in \mathbb{R}$ :
  \[
    e^{i\theta} = \cos\theta + i\sin\theta.
  \]
  La forme exponentielle est donc $z = |z|\,e^{i\arg z}$.
\end{tcolorbox}

\rubriq{Multiplication en forme polaire}

\begin{mdframed}[style=proofbar]
  Si $z = r e^{i\theta}$ et $w = s e^{i\varphi}$, alors
  \[
    zw = rs\, e^{i(\theta + \varphi)}.
  \]
  \textit{RĂ¨gles pour l'argument :}
  $\arg(zw) \equiv \arg z + \arg w \pmod{2\pi}$,
  $\arg(z^{-1}) \equiv -\arg z$,
  $\arg(\bar{z}) \equiv -\arg z$.

  \textit{ConsĂŠquence gĂŠomĂŠtrique :} multiplier par $e^{i\alpha}$ est une
  \textbf{rotation}\index{rotation (multiplication complexe)} de centre $O$ et d'angle $\alpha$ dans le plan complexe.
  Multiplier par $r > 0$ est une homothĂŠtie de rapport $r$. \hfill$\blacksquare$
\end{mdframed}

\rubriq{Formules de De Moivre et de linĂŠarisation}

\begin{tcolorbox}[thmbox]
  \textbf{Formule de De Moivre.}\enspace%
  \index{De Moivre (formule de)}
  Pour tout $n \in \mathbb{Z}$ et $\theta \in \mathbb{R}$ :
  \[
    (\cos\theta + i\sin\theta)^n = \cos(n\theta) + i\sin(n\theta).
  \]

  \medskip
  \textbf{Formules d'Euler inverses.}\enspace%
  \[
    \cos\theta = \frac{e^{i\theta} + e^{-i\theta}}{2}, \qquad
    \sin\theta = \frac{e^{i\theta} - e^{-i\theta}}{2i}.
  \]
  Elles permettent de \textbf{linĂŠariser}\index{linĂŠarisation trigonomĂŠtrique} $\cos^n\theta$ et $\sin^n\theta$
  (exprimer une puissance en combinaison de $\cos(k\theta)$, $\sin(k\theta)$).
\end{tcolorbox}

\begin{mdframed}[style=proofbar]
  \textit{Preuve de De Moivre.} Par rĂŠcurrence sur $n \geq 0$ en utilisant
  $e^{i(n+1)\theta} = e^{in\theta} \cdot e^{i\theta}$. Le cas $n < 0$ suit
  par passage Ă  l'inverse. \hfill$\blacksquare$
\end{mdframed}

\decsep{}

\secnum{navyblue}{3}{Racines $n$-iĂ¨mes}

\begin{tcolorbox}[thmbox]
  \textbf{ThĂŠorĂ¨me (racines $n$-iĂ¨mes).}\enspace%
  \index{racines n-iĂ¨mes}
  Soit $n \geq 1$ et $w = \rho e^{i\varphi} \in \mathbb{C}^*$. L'ĂŠquation
  $z^n = w$ admet exactement $n$ solutions dans $\mathbb{C}$ :
  \[
    z_k = \rho^{1/n}\, \exp\!\left(i\,\frac{\varphi + 2k\pi}{n}\right),
    \quad k = 0, 1, \ldots, n-1.
  \]
  Ces solutions sont les sommets d'un polygone rĂŠgulier Ă  $n$ cĂ´tĂŠs inscrit
  dans le cercle de rayon $\rho^{1/n}$, ĂŠquidistants d'angle $2\pi/n$.

  \medskip
  \textbf{Racines de l'unitĂŠ.}\enspace%
  \index{racines de l'unitĂŠ}
  Pour $w = 1$ ($\rho = 1$, $\varphi = 0$) :
  \[
    \omega_k = e^{2ik\pi/n}, \quad k = 0, \ldots, n-1.
  \]
  Ces racines vĂŠrifient $\displaystyle\sum_{k=0}^{n-1} \omega_k^j = 0$
  pour $j \not\equiv 0 \pmod{n}$.
\end{tcolorbox}

\begin{mdframed}[style=proofbar]
  \textit{Existence.} On vĂŠrifie $z_k^n = \rho\,e^{i(\varphi+2k\pi)} = \rho e^{i\varphi} = w$.

  \textit{UnicitĂŠ (exactement $n$ solutions).} Si $z^n = w$, on ĂŠcrit $z = r e^{i\alpha}$.
  Alors $r^n = \rho$ donne $r = \rho^{1/n}$ (unique rĂŠel positif), et
  $n\alpha \equiv \varphi \pmod{2\pi}$, ce qui donne exactement $n$ valeurs
  $\alpha \in [0, 2\pi[$.

  \textit{Somme des racines.} Pour $j \not\equiv 0 \pmod{n}$, la somme est
  une sĂŠrie gĂŠomĂŠtrique :
  $\sum_{k=0}^{n-1} \omega_k^j = \sum_{k=0}^{n-1} (e^{2i\pi j/n})^k
  = \dfrac{1 - e^{2i\pi j}}{1 - e^{2i\pi j/n}} = 0$.
  \hfill$\blacksquare$
\end{mdframed}

\secnum{navyblue}{4}{GĂŠomĂŠtrie du plan complexe}

\begin{tcolorbox}[thmbox]
  \textbf{Transformations gĂŠomĂŠtriques.}\enspace%
  \index{plan complexe!transformations}
  On identifie $\mathbb{C}$ et $\mathbb{R}^2$ via $z = x + iy \leftrightarrow (x,y)$.
  \begin{itemize}
    \item \textbf{Translation} de vecteur $b$ : $z \mapsto z + b$.
    \item \textbf{HomothĂŠtie} de centre $O$, rapport $r > 0$ : $z \mapsto rz$.
    \item \textbf{Rotation} de centre $O$, angle $\alpha$ : $z \mapsto e^{i\alpha} z$.
    \item \textbf{Similitude directe} (gĂŠnĂŠrale) : $z \mapsto az + b$,
          $a \in \mathbb{C}^*$, $b \in \mathbb{C}$.
    \item \textbf{RĂŠflexion} d'axe rĂŠel : $z \mapsto \bar{z}$.
    \item \textbf{Inversion} de centre $O$, puissance $1$ : $z \mapsto 1/\bar{z}$.
  \end{itemize}
\end{tcolorbox}

\rubriq{Alignement et cocyclicitĂŠ}

\begin{mdframed}[style=proofbar]
  Trois points $A$, $B$, $C$ d'affixes $a$, $b$, $c$ sont
  \textbf{alignĂŠs}\index{alignement (plan complexe)} si et seulement si
  \[
    \frac{c - a}{b - a} \in \mathbb{R}.
  \]
  Quatre points sont \textbf{cocycliques ou alignĂŠs}\index{cocyclicitĂŠ} si et
  seulement si leur birapport\index{birapport}
  $[z_1, z_2; z_3, z_4] = \dfrac{(z_3-z_1)(z_4-z_2)}{(z_3-z_2)(z_4-z_1)}$
  est rĂŠel. \hfill$\blacksquare$
\end{mdframed}
